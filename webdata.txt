function map(id) {
  id = id.replace('#', '');

  var parts = id.split('/');
  var len = parts.length;
  var root, name;

  // id = root/name/x.y.z/name
  if (len === 4) {
    root = parts[0];
    name = parts[1];

    // gallery 或 alipay 开头的没有问题
    if (root === 'alipay' || root === 'gallery') {
      return id;
    }

    // arale 开头的

      // 处理 arale/handlebars 的情况

        return id.replace('arale/', 'gallery/');

    name = parts[0]

      return 'gallery/' + id;

      return 'arale/' + id;
